15  /**
16  * \ingroup threading
17  *
18  * Simple wrapper around pthread's mutex.
19  * Before you use, see \ref parallel_object_intricacies.
20  *
21  * Windows recursive mutex are annoyingly recursive.
22  * We need to prevent recursive locks. We do this by associating an
23  * addition boolean "locked" to the mutex.
24  * Hence in the event of a double lock, on Windows the behavior is slightly
25  * different. On Linux/Mac this will trigger a deadlock. On Windows,
26  * this will trigger an assertion failure.
27  */
28  class mutex {
29  public:
30  // mutable not actually needed
31  mutable pthread_mutex_t m_mut;
32 #ifdef _WIN32
33  mutable volatile bool locked = false;
34 #endif
35  /// constructs a mutex
36  mutex() {
37  int error = pthread_mutex_init(&m_mut, NULL);
38  ASSERT_MSG(!error, "Mutex create error %d", error);
39  }
40  /** Copy constructor which does not copy. Do not use!
41  Required for compatibility with some STL implementations (LLVM).
42  which use the copy constructor for vector resize,
43  rather than the standard constructor. */
44  mutex(const mutex&) {
45  int error = pthread_mutex_init(&m_mut, NULL);
46  ASSERT_MSG(!error, "Mutex create error %d", error);
47  }
48 
49  ~mutex() noexcept {
50  int error = pthread_mutex_destroy( &m_mut );
51  if(error) {
52  try {
53  std::cerr << "Mutex destroy error " << error << std::endl;
54  } catch (...) {
55  }
56  abort();
57  }
58  }
59 
60  // not copyable
61  void operator=(const mutex& m) { }
62 
63  /// Acquires a lock on the mutex
64  inline void lock() const {
65  TURI_ATTRIBUTE_UNUSED_NDEBUG int error = pthread_mutex_lock( &m_mut );
66  DASSERT_MSG(!error, "Mutex lock error %d", error);
67 #ifdef _WIN32
68  DASSERT_TRUE(!locked);
69  locked = true;
70 #endif
71  }
72  /// Releases a lock on the mutex
73  inline void unlock() const {
74 #ifdef _WIN32
75  locked = false;
76 #endif
77 
78  TURI_ATTRIBUTE_UNUSED_NDEBUG int error = pthread_mutex_unlock( &m_mut );
79  DASSERT_MSG(!error, "Mutex unlock error %d", error);
80  }
81  /// Non-blocking attempt to acquire a lock on the mutex
82  inline bool try_lock() const {
83 #ifdef _WIN32
84  if (locked) return false;
85 #endif
86  return pthread_mutex_trylock( &m_mut ) == 0;
87  }
88  friend class conditional;
89  }; // End of Mutex
