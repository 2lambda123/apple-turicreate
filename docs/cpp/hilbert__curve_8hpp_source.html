23 /**
24  * For an n*n square and a hilbert index (s) ranging from 0 to n*n-1, this
25  * function returns the coordinate of the s^th position along the hilbert
26  * curve. n must be at least 2 and must be a power of 2.
27  *
28  * Algorithm from Figure 14-8 in Hacker's Delight.
29  */
30 inline std::pair<size_t, size_t> hilbert_index_to_coordinate(size_t s, size_t n) {
31  if (s == 0 && n == 1) return {0, 0};
32 
33  ASSERT_GE(n, 2);
34  ASSERT_TRUE(is_power_of_2(n));
35  n = __builtin_ctz(n); // convert to the "Order" of the curve. i.e. log(n)
36  size_t comp, swap, cs, t, sr;
37 
38  s = s | (0x55555555 << 2*n); // Pad s on left with 01
39  sr = (s >> 1) & 0x55555555; // (no change) groups.
40  cs = ((s & 0x55555555) + sr) // Compute complement &
41  ^ 0x55555555; // swap info in two-bit
42  // groups.
43  // Parallel prefix xor op to propagate both complement
44  // and swap info together from left to right (there is
45  // no step "cs ^= cs >> 1", so in effect it computes
46  // two independent parallel prefix operations on two
47  // interleaved sets of sixteen bits).
48 
49  cs = cs ^ (cs >> 2);
50  cs = cs ^ (cs >> 4);
51  cs = cs ^ (cs >> 8);
52  cs = cs ^ (cs >> 16);
53  swap = cs & 0x55555555; // Separate the swap and
54  comp = (cs >> 1) & 0x55555555; // complement bits.
55 
56  t = (s & swap) ^ comp; // Calculate x and y in
57  s = s ^ sr ^ t ^ (t << 1); // the odd & even bit
58  // positions, resp.
59  s = s & ((1 << 2*n) - 1); // Clear out any junk
60  // on the left (unpad).
61 
62  // Now "unshuffle" to separate the x and y bits.
63 
64  t = (s ^ (s >> 1)) & 0x22222222; s = s ^ t ^ (t << 1);
65  t = (s ^ (s >> 2)) & 0x0C0C0C0C; s = s ^ t ^ (t << 2);
66  t = (s ^ (s >> 4)) & 0x00F000F0; s = s ^ t ^ (t << 4);
67  t = (s ^ (s >> 8)) & 0x0000FF00; s = s ^ t ^ (t << 8);
68  return {s >> 16, s & 0xFFFF};
69 }

72 /**
73  * For an n*n square and a coordinate within the square. Returns the
74  * hilbert index which is the position of the coordinate along the hilbert curve.
75  * n must be at least 2 and must be a power of 2.
76  *
77  * Algorithm from Figure 14-9 in Hacker's Delight.
78  */
79 inline size_t coordinate_to_hilbert_index(std::pair<size_t, size_t> coord, size_t n) {
80  if ((n == 1) && (coord.first == 0) && (coord.second = 0)) return 0;
81 
82  ASSERT_GE(n, 2);
83  ASSERT_TRUE(is_power_of_2(n));
84  ASSERT_LT(coord.first, n);
85  ASSERT_LT(coord.second, n);
86 
87  n = __builtin_ctz(n); // convert to the "Order" of the curve. i.e. log(n)
88  int i;
89  size_t state, s, row;
90  size_t x = coord.first;
91  size_t y = coord.second;
92 
93  state = 0; // Initialize.
94  s = 0;
95  for (i = n - 1; i >= 0; i--) {
96  row = 4*state | 2*((x >> i) & 1) | ((y >> i) & 1);
97  s = (s << 2) | ((0x361E9CB4 >> 2*row) & 3);
98  state = (0x8FE65831 >> 2*row) & 3;
99  }
100  return s;
101 }
