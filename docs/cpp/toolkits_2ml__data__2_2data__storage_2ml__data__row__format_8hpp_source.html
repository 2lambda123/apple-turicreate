21 /**
22  * ml_data layout format
23  * ================================================================================
24  *
25  * The rows are stored in segments of a vector of entry_value structs,
26  * where entry_value encloses a union of an index_value and a
27  * double_value. Thus it is 8 bytes. Each vector contains up to
28  * row_block_size rows. The data is stored as an sarray<vector<entry_value> >
29  *
30  * Raw data layout
31  * ---------------
32  *
33  * Each vector is simply laid out as
34  *
35  * | row 1 | row 2 | ... | row n |
36  *
37  * where n <= row_block_size.
38  *
39  * Each row is laid out according to the "mode" of the columns,
40  * determined by column_metadata::mode. mode can be NUMERIC,
41  * CATEGORICAL, NUMERIC_VECTOR, CATEGORICAL_VECTOR, and7 DICTIONARY. The
42  * last three of these can hold multiple values.
43  *
44  * ROW FORMAT (from ml_data.hpp)
45  * ================================================================================
46  *
47  * ml_data loads data from an existing sframe, indexes it by mapping all
48  * categorical values to unique indices in 0, 1,2,...,n, and records
49  * statistics about the values. It then puts it into an efficient
50  * row-based data storage structure for use in regression and learning
51  * algorithms. The row based storage structure is designed for fast
52  * iteration through the rows and target.
53  *
54  *
55  * The rows are stored in segments of a vector of entry_value structs,
56  * where entry_value encloses a union of an index_value and a
57  * double_value. Thus it is 8 bytes. Each vector contains up to
58  * ROW_BLOCK_SIZE rows. The data is stored as an sarray<vector<entry_value> >
59  *
60  * Raw data layout
61  * ---------------
62  *
63  * Each vector is simply laid out as
64  *
65  * | <row_size> row 1 | <row_size> row 2 | ... | <row_size> row n |
66  *
67  * The row size variable is present if CATEGORICAL_VECTOR or
68  * DICTIONARY are defined types, as these may induce variable sized
69  * rows. Otherwise, the row sizes are constant and determined from
70  * the metadata, thus this does not need to be there.
71  *
72  * where n <= ROW_BLOCK_SIZE.
73  *
74  * Each row is laid out according to the "mode" of the columns,
75  * determined by column_metadata::mode. mode can be NUMERIC,
76  * CATEGORICAL, NUMERIC_VECTOR, CATEGORICAL_VECTOR, and DICTIONARY. The
77  * last three of these can hold multiple values.
78  *
79  * Row layout
80  * ----------
81  *
82  * Each row takes a block of data entries laid out according to mode:
83  *
84  * NUMERIC:
85  *
86  * ... | value | ...
87  *
88  * CATEGORICAL:
89  *
90  * ... | index | ...
91  *
92  * NUMERIC_VECTOR:
93  *
94  * ... | value1 value2 ... | ...
95  *
96  * CATEGORICAL_VECTOR:
97  *
98  * ... | length index1 index2 ... | ...
99  *
100  * DICTIONARY:
101  *
102  * ... | length index1 value1 index2 value2 ... | ...
103  *
104  * example: {0 : 2.1, 1 : 8.5} would be ... | 2 0 2.1 1 8.5 | ...
105  *
106  *
107  * Since the number of columns, and the mode of each column is constant
108  * across rows, the values are all laid out sequentially; for example, a
109  * row with columns (numeric, numeric_vector, dictionary) would be
110  *
111  * ... || row_size | value | v1 v2 ... | length k1 v1 k2 v2 ... || ...
112  *
113  * Thus a row of [1.0, [2.0, 4.0], {0 : 2.1, 1 : 8.5}] would appear as
114  *
115  * ... || 8 | 1.0 | 2.0 4.0 | 2 0 2.1 1 8.5 || ...
116  * = ... || 8 1.0 2.0 4.0 2 0 2.1 1 8.5 || ...
117  *
118  * Similarly, a row with columns (numeric, numeric_vector, categorical)
119  * would be
120  *
121  * ... || value | v1 v2 ... | k1 || ...
122  * * And a row of [1.0, [2.0, 4.0], 8] would appear as
123  *
124  * ... || 1.0 | 2 2.0 4.0 | 8 || ...
125  * = ... || 1.0 2 2.0 4.0 8 || ...
126  * Since ROW_BLOCK_SIZE are stored together in a vector<entry_value>,
127  * iterating through rows is extremely fast.
128  *
129  *
130  * target access
131  * ======================================================================
132  *
133  * If a target column is present, it is stored as the LAST element in
134  * exactly the same way as a regular numeric value.
135  */

312 /** Reads the rows of data out as a sequence of (column_index,
313  * feature_index, double value) tuples.
314  *
315  * EntryWriteFunction out_function is a function taking 6 arguments:
316  *
317  * (ml_column_mode mode,
318  * size_t column_index,
319  * size_t feature_index,
320  * double value,
321  * size_t index_size,
322  * size_t index_offset).
323  *
324  * row_block_ptr is a pointer or iterator to the first entry_value of
325  * the row.
326  *
327  * rm is the row_metadata structure associated with this row.
328  */
329 template <
330  typename EntryWriteFunction,
331  typename ColumnAdvancementFunction>
332 GL_HOT_INLINE_FLATTEN
333 static inline void
334 _read_raw_ml_data_column_block(
335  const row_metadata& rm,
336  entry_value_iterator& row_block_ptr,
337  EntryWriteFunction out_function,
338  ColumnAdvancementFunction next_column) {
339 
340 #ifndef NDEBUG
341  entry_value_iterator original_row_block_ptr = row_block_ptr;
342  entry_value_iterator row_block_end = row_block_ptr + get_row_data_size(rm, row_block_ptr);
343 #endif
344 
345  DASSERT_EQ(rm.num_x_columns + (rm.has_target ? 1 : 0), rm.total_num_columns);
346 
347  /** Skip the first entry if the row size is not constant, as this
348  * contains the row data size. */
349  if(!rm.data_size_is_constant)
350  ++row_block_ptr;
351 
352  const size_t num_columns = rm.num_x_columns;
353 
354  for(size_t c_idx = 0; c_idx < num_columns; ++c_idx) {
355 
356  DASSERT_LT(c_idx, rm.metadata_vect.size());
357 
358  ////////////////////////////////////////////////////////////////////////////////
359  // Step 1: Set up everything needed to read out values from the raw data.
360 
361  size_t index_size = rm.metadata_vect[c_idx]->index_size();
362  size_t index_offset = rm.metadata_vect[c_idx]->global_index_offset();
363 
364  // The offset should always have been initialized.
365  DASSERT_NE(index_offset, size_t(-1));
366 
367  /** Call this to read out an index.
368  */
369  auto read_index = [&]() GL_GCC_ONLY(GL_HOT_INLINE_FLATTEN) {
370  DASSERT_LT(row_block_ptr, row_block_end);
371 
372  size_t index = row_block_ptr->index_value;
373  ++row_block_ptr;
374  return index;
375  };
376 
377  /** Call this to read out a value.
378  */
379  auto read_value = [&]() GL_GCC_ONLY(GL_HOT_INLINE_FLATTEN) {
380  DASSERT_LT(row_block_ptr, row_block_end);
381 
382  double value = row_block_ptr->double_value;
383  ++row_block_ptr;
384  return value;
385  };
386 
387  /** Call this to read out a size.
388  */
389  auto read_size = [&]() GL_GCC_ONLY(GL_HOT_INLINE_FLATTEN) {
390  size_t size = row_block_ptr->index_value;
391  ++row_block_ptr;
392 
393  DASSERT_FALSE(rm.data_size_is_constant);
394  DASSERT_LT(size, get_row_data_size(rm, original_row_block_ptr));
395 
396  return size;
397  };
398 
399  ////////////////////////////////////////////////////////////////////////////////
400  // Step 2: Go through and read values based on the type.
401 
402  // For some of the modes, an important optimization is to set
403  // all_columns_categorical to be true in the data profile, which
404  // means that this branch / lookup is optimized out at compile
405  // time.
406 
407  switch(rm.metadata_vect[c_idx]->mode) {
408 
409  case ml_column_mode::NUMERIC: {
410 
411  size_t index = 0;
412  double value = read_value();
413 
414  out_function(ml_column_mode::NUMERIC, c_idx, index, value, index_size, index_offset);
415  next_column(ml_column_mode::NUMERIC, c_idx, index_size);
416 
417  break;
418  }
419 
420  case ml_column_mode::CATEGORICAL: {
421 
422  // Just in case we use the all_columns_categorical override --
423  // check to make sure it's valid.
424  DASSERT_TRUE(rm.metadata_vect[c_idx]->mode == ml_column_mode::CATEGORICAL);
425 
426  size_t index = read_index();
427  double value = 1.0;
428 
429  out_function(ml_column_mode::CATEGORICAL, c_idx, index, value, index_size, index_offset);
430  next_column(ml_column_mode::CATEGORICAL, c_idx, index_size);
431  break;
432  }
433 
434  case ml_column_mode::NUMERIC_VECTOR: {
435 
436  size_t nv = rm.metadata_vect[c_idx]->fixed_column_size();
437 
438  for(size_t k = 0; k < nv; ++k) {
439 
440  size_t index = k;
441  double value = read_value();
442 
443  out_function(ml_column_mode::NUMERIC_VECTOR, c_idx, index, value, index_size, index_offset);
444  }
445 
446  next_column(ml_column_mode::NUMERIC_VECTOR, c_idx, index_size);
447  break;
448  }
449 
450  case ml_column_mode::CATEGORICAL_VECTOR: {
451 
452  size_t nv = read_size();
453 
454  for(size_t k = 0; k < nv; ++k) {
455 
456  size_t index = read_index();
457  double value = 1.0;
458 
459  out_function(ml_column_mode::CATEGORICAL_VECTOR, c_idx, index, value, index_size, index_offset);
460  }
461 
462  next_column(ml_column_mode::CATEGORICAL_VECTOR, c_idx, index_size);
463  break;
464  }
465 
466  case ml_column_mode::DICTIONARY: {
467 
468  size_t nv = read_size();
469 
470  for(size_t k = 0; k < nv; ++k) {
471 
472  size_t index = read_index();
473  double value = read_value();
474 
475  out_function(ml_column_mode::DICTIONARY, c_idx, index, value, index_size, index_offset);
476  }
477 
478  next_column(ml_column_mode::DICTIONARY, c_idx, index_size);
479  break;
480  }
481 
482  case ml_column_mode::UNTRANSLATED: {
483 
484  // Do nothing in this case.
485  break;
486  }
487  default:
488  DASSERT_TRUE(false);
489  } // End switch
490  }
491 }

494 /** Reads the rows of data out as a sequence of (column_index,
495  * feature_index, double value) tuples.
496  *
497  * EntryWriteFunction out_function is a function taking arguments (size_t column_index,
498  * size_t feature_index, double value).
499  *
500  * row_block_ptr is a pointer or iterator to the first entry_value of
501  * the row.
502  *
503  * rm is the row_metadata structure associated with this row.
504  */
505 template <typename EntryWriteFunction,
506  typename ColumnAdvancementFunction,
507  typename MLDataSideFeaturesType> // Use a template here as
508  // a cheap way to avoid
509 // cyclical dependencies.
510 GL_HOT_INLINE_FLATTEN
511 static inline void
512 read_ml_data_row(
513  const row_metadata& rm,
514  entry_value_iterator row_block_ptr,
515  EntryWriteFunction out_function,
516  ColumnAdvancementFunction next_column,
517  const std::shared_ptr<MLDataSideFeaturesType>& side_features) {
518 
519 #ifndef NDEBUG
520  size_t row_data_size = get_row_data_size(rm, row_block_ptr);
521 #endif
522 
523  const bool use_side_features = (side_features != nullptr);
524  const entry_value_iterator row_block_start = row_block_ptr;
525 
526  // Here, we've got num_x_columns values for x, with the first one
527  // for y.
528  {
529  auto out_function_with_hook =
530  [&](ml_column_mode mode, size_t column_index,
531  size_t feature_index, double value,
532  size_t index_size, size_t index_offset) GL_GCC_ONLY(GL_HOT_INLINE_FLATTEN) {
533 
534  out_function(mode, column_index, feature_index, value, index_size, index_offset);
535  };
536 
537  _read_raw_ml_data_column_block(rm, row_block_ptr, out_function_with_hook, next_column);
538  }
539 
540 
541  // Some functional code, pun intended.
542  if(use_side_features) {
543 
544  // Loop through the columns relevant to the side data; this needs
545  // to be done by supplying the function called at each entry,
546  // since we'll just read that data again.
547 
548  // Call this code as it's own function since side data is not
549  // commonly used.
550  auto add_side_features_sep_function = [&]() GL_GCC_ONLY(GL_HOT_NOINLINE_FLATTEN) {
551 
552  auto out_function_to_add_side_data = [&](
553  ml_column_mode mode, size_t column_index,
554  size_t feature_index, double value,
555  size_t index_size, size_t index_offset) GL_GCC_ONLY(GL_HOT_INLINE_FLATTEN) {
556 
557  if(mode == ml_column_mode::CATEGORICAL) {
558 
559  // Get the data information for this particular column.
560  auto data = side_features->get_side_feature_block(column_index, feature_index);
561 
562  // Case 1: nothing in the side data for this column / feature.
563  if(data.row_block_ptr == nullptr) {
564 
565  // If there is nothing in the side data for this feature,
566  // we still need to call next_column on these so the
567  // function calling this can maintain state.
568  for(size_t i = 0; i < data.rm.num_x_columns; ++i) {
569  next_column(data.rm.metadata_vect[i]->mode,
570  i + data.column_offset,
571  data.rm.metadata_vect[i]->index_size());
572  }
573 
574  // Case 2: join on the data.
575  } else {
576 
577  // There is legit data in this; we just need to call
578  // out_function on every element now. However, the
579  // column_offset needs to match the overall columns, not
580  // the ones in data.rm (which is just local to the side
581  // info), so hook the user-supplied out_function and
582  // offset the column index.
583 
584  auto sd_out_function_with_hook =
585  [&](ml_column_mode mode, size_t column_index,
586  size_t feature_index, double value,
587  size_t index_size, size_t index_offset)
588  GL_GCC_ONLY(GL_HOT_INLINE_FLATTEN) {
589 
590  // Call the user-supplied out_function with this side
591  // column's entry. Note that column_index here is in
592  // reference to data.rm, which is this side column's
593  // data, not the overall column index; thus shift it by
594  // the offset.
595  out_function(mode, data.column_offset + column_index,
596  feature_index, value,
597  index_size, index_offset);
598  };
599 
600  // Read this raw data into the user's out_function
601  // (through sd_out_function_with_hook) and next_column.
602  _read_raw_ml_data_column_block(
603  data.rm, data.row_block_ptr, sd_out_function_with_hook, next_column);
604  }
605  }
606  };
607 
608  entry_value_iterator sd_row_block_ptr = row_block_start;
609 
610  // Now, reread the original data to get the column indices of
611  // on which the side_features have to be joined.
612  _read_raw_ml_data_column_block(
613  rm, sd_row_block_ptr, out_function_to_add_side_data,
614  [](ml_column_mode, size_t, size_t){});
615  };
616 
617  // Call the above non-inlined function.
618  add_side_features_sep_function();
619  }
620 
621  // Check to make sure the layout is good. Note that the
622  // row_data_size, if not constant, does not include it's own index.
623 
624 #ifndef NDEBUG
625  if(rm.has_target)
626  ++row_block_ptr;
627 
628  DASSERT_EQ(std::distance(row_block_start, row_block_ptr), row_data_size);
629 #endif
630 }
631
