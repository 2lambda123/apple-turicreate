177 /**
178  * \ingroup fileio
179  * Given a root directory and an absolute path, tries to create a relative path
180  * address between root_directory and the path; if not possible, returns the
181  * the original path with no changes.
182  *
183  * This function is relatively limited. It will not add "../" structures to the
184  * returned relative path. In other words, the path must point to a file/folder
185  * inside of the root directory for this to return a relative path.
186  *
187  * Example:
188  * make_relative_path("s3://bucket/data", "s3://bucket/data/123")
189  * returns "123".
190  *
191  * make_relative_path("s3://bucket/data", "s3://foo/123")
192  * returns "s3://foo/123".
193  */
194 std::string make_relative_path(std::string root_directory, std::string path);

196 /**
197  * \ingroup fileio
198  * Given a root directory and a relative path, tries to convert the relative
199  * path to an absolute path. If the path is already an absolute path, returns
200  * the original path with no changes.
201  *
202  * This function is absolutely limited. It will not handle "../" structures to
203  * the returned relative path. In other words, the output path path must point
204  * to a file/folder inside of the root directory.
205  *
206  * Example:
207  * make_absolute_path("s3://bucket/data", "123")
208  * returns "s3://bucket/data/123".
209  *
210  * make_absolute_path("s3://bucket/data", "s3://foo/123")
211  * returns "s3://foo/123".
212  */
