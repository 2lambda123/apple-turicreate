27 class opt_logical_filter_linear_transform_exchange
28  : public opt_logical_filter_transform {
29 
30  std::string description() {
31  return "logical_filter(linear_transform(a), b) -> linear_transform(logical_filter(a, b))";
32  }
33 
34  // Should include union nodes as well.
35  bool apply_transform(optimization_engine *opt_manager, cnode_info_ptr n) {
36  DASSERT_TRUE(n->type == planner_node_type::LOGICAL_FILTER_NODE);
37 
38  // Activated only if it's a linear transform, but not a project
39  // node, and only if there is one output.
40 
41  if(!n->inputs[0]->is_linear_transform()
42  || n->inputs[0]->outputs.size() > 1
43  || n->inputs[0]->type == planner_node_type::PROJECT_NODE)
44  return false;
45 
46  pnode_ptr ret = n->inputs[0]->pnode->clone();
47  ret->inputs.resize(n->inputs[0]->pnode->inputs.size());
48 
49  pnode_ptr logical_filter_mask = n->inputs[1]->pnode;
50 
51  for(size_t i = 0; i < n->inputs[0]->pnode->inputs.size(); ++i) {
52  ret->inputs[i] = op_logical_filter::make_planner_node(n->inputs[0]->pnode->inputs[i], logical_filter_mask);
53  }
54 
55  opt_manager->replace_node(n, ret);
56  return true;
57  }
58 };

60 class opt_logical_filter_expanding_project_exchange
61  : public opt_logical_filter_transform {
62 
63  std::string description() {
64  return "logical_filter(expanding_project(a), b) -> expanding_project(logical_filter(a, b))";
65  }
66 
67  // Move expanding projects through the logical filter.
68  bool apply_transform(optimization_engine *opt_manager, cnode_info_ptr n) {
69  DASSERT_TRUE(n->type == planner_node_type::LOGICAL_FILTER_NODE);
70 
71  // Activated only if it's a linear transform, but not a project
72  // node, and only if there is one output.
73 
74  if(n->inputs[0]->type != planner_node_type::PROJECT_NODE)
75  return false;
76 
77  // Determine what kind of transformation we can do.
78  const auto& iv = n->inputs[0]->p("indices").get<flex_list>();
79 
80  // If it doesn't expand things, then don't do an exchange.
81  if(iv.size() <= n->inputs[0]->inputs[0]->num_columns())
82  return false;
83 
84  pnode_ptr new_fltr = op_logical_filter::make_planner_node(n->inputs[0]->inputs[0]->pnode, n->inputs[1]->pnode);
85 
86  pnode_ptr new_proj = n->inputs[0]->pnode->clone();
87  new_proj->inputs = {new_fltr};
88 
89  opt_manager->replace_node(n, new_proj);
90  return true;
91  }

94 class opt_merge_identical_logical_filters
95  : public opt_logical_filter_transform {
96 
97  std::string description() {
98  return "logical_filter(b, a), ..., logical_filter(c, a) -> logical_filter(..., a)";
99  }
100 
101  // Should include union nodes as well.
102  bool apply_transform(optimization_engine *opt_manager, cnode_info_ptr n) {
103  DASSERT_TRUE(n->type == planner_node_type::LOGICAL_FILTER_NODE);
104 
105  // Activated only if it's a linear transform, but not a project
106  // node, and only if there is one output.
107 
108  size_t n_logical_filter_outs = 0;
109  for(const cnode_info_ptr& nn : n->inputs[1]->outputs) {
110  if(nn->type == planner_node_type::LOGICAL_FILTER_NODE
111  && nn->inputs[1] == n->inputs[1]) {
112 
113  ++n_logical_filter_outs;
114  }
115  }
116 
117  if(n_logical_filter_outs <= 1)
118  return false;
119 
120  std::vector<pnode_ptr> union_inputs(n_logical_filter_outs);
121  std::vector<std::vector<size_t> > projection_outputs(n_logical_filter_outs);
122  std::vector<cnode_info_ptr> rep_nodes(n_logical_filter_outs);
123 
124  size_t idx = 0;
125  size_t out_idx = 0;
126  for(const cnode_info_ptr& nn : n->inputs[1]->outputs) {
127  if(nn->type == planner_node_type::LOGICAL_FILTER_NODE
128  && nn->inputs[1] == n->inputs[1]) {
129 
130  union_inputs[idx] = nn->inputs[0]->pnode;
131 
132  projection_outputs[idx].resize(nn->num_columns());
133  std::iota(projection_outputs[idx].begin(), projection_outputs[idx].end(), out_idx);
134  out_idx += nn->num_columns();
135 
136  rep_nodes[idx] = nn;
137 
138  ++idx;
139  }
140  }
141 
142  // Should always be true, as there can't really be duplicates here
143 #ifndef NDEBUG
144  {
145  size_t n_unique = std::set<cnode_info_ptr>(rep_nodes.begin(), rep_nodes.end()).size();
146  DASSERT_EQ(n_unique, n_logical_filter_outs);
147  }
148 #endif
149 
150  pnode_ptr main_in = op_union::make_planner_node(union_inputs);
151  pnode_ptr filter = op_logical_filter::make_planner_node(main_in, n->inputs[1]->pnode);
152 
153  for(size_t i = 0; i < n_logical_filter_outs; ++i) {
154  pnode_ptr new_pr_out = op_project::make_planner_node(filter, projection_outputs[i]);
155  opt_manager->replace_node(rep_nodes[i], new_pr_out);
156  }
157 
158  return true;
159  }
160 };
