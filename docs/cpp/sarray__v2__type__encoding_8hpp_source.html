123 /**
124  * Encodes a collection of flexible_type values. The array must be of
125  * contiguous type, but permitting undefined values.
126  *
127  * There is a two byte header to the block.
128  * - num_types: 1 byte
129  * - if 0, the block is empty.
130  * - if 1, the array is of contiguous type (see next byte)
131  * - if 2, the array is of contiguous type, but has missing values.
132  * - type: 1 byte.
133  * - [undefined bitfield]: if type is 2, this contains a bitfield of
134  * (round_op(#elem / 8) bytes) listing the positions of all the UNDEFINED
135  * fields)
136  * - type specific encoding:
137  * - if integer or float, encode_number() is called
138  * - if string, encode_string() is called
139  * - otherwise, direct serialization is currently used.
140  * - If UNDEFINED (i.e. array is of all UNDEFINED values, nothing is written)
141  *
142  * \note The coding does not store the number of values stored. This is
143  * stored in the block_info (block.num_elem)
144  */
145 void typed_encode(const std::vector<flexible_type>& data,
146  block_info& info,
147  oarchive& oarc);
