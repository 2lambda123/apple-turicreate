68  /** Defines the output of a data_iterator. */
69  struct batch {
70 
71  /** Defines the metadata associated with each chunk. */
72  struct chunk_info {
73 
74  /** The session ID from which the chunk was segmented. */
75  flexible_type session_id;
76 
77  /** Number of samples (rows from the raw SFrame) comprising the chunk. */
78  size_t num_samples;
79  };
80 
81  /**
82  * An array with shape: (requested_batch_size,
83  * 1, prediction_window * predictions_in_chunk, num_feature_columns)
84  *
85  * Each row is a chunk of feature values from one session.
86  */
87  neural_net::shared_float_array features;
88 
89  /**
90  * An array with shape: (requested_batch_size, 1, predictions_in_chunk, 1)
91  *
92  * Each row is the sequence of class label (indices) from one chunk.
93  *
94  * If no target was specified, then this value is default constructed.
95  */
96  neural_net::shared_float_array labels;
97 
98  /**
99  * An array with shape: (requested_batch_size, 1, predictions_in_chunk, 1)
100  *
101  * Each row is a sequence of 0 or 1 values indicating whether the
102  * corresponding label is padding (0) or refers to actual data (1).
103  *
104  * If no target was specified, then this value is default constructed.
105  */
106  neural_net::shared_float_array weights;
107 
108  /**
109  * Metadata for each valid (non-padded) row in the batch.
110  *
111  * The size of this vector is at most requested_batch_size. The info at
112  * index `i` describes the array at `features[i]`.
113  */
114  std::vector<chunk_info> batch_info;
115  };

129  /**

184 /**
185  * Convert SFrame to batch form, where each row contains a sequence of length
186  * predictions_in_chunk * prediction_window, and there is a single label per
187  * prediction window.
188  *
189  * \param[in] data Original data. Sframe containing one line per time sample.
190  * \param[in] features List of names of the columns containing the input features.
191  * \param[in] session_id Name of the column containing ids for each session in the dataset.
192  * A session is a single user time-series sequence.
193  * \param[in] prediction_window Number of time samples in every prediction window. A label is expected
194  * (for training), or predicted (in inference) every time a sequence of
195  * prediction_window samples have been collected.
196  * \param[in] predictions_in_chunk Each session is chunked into shorter sequences. This is the number of
197  * prediction windows desired in each chunk.
198  * \param[in] target Name of the coloumn containing the output labels. Empty string if None.
199  *
200  * \return SFrame with the data converted to batch form.
201  */
