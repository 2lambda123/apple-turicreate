44 /**
45  * Copies a row into a sequence of ml_data_entry structures sent to
46  * the output iterator MLDataEntryIterator x_out. This provides fast
47  * translation between the row blocks and ml_data_entry structures.
48  *
49  * \param[in,out] row_block_ptr Points to the beginning of the row in
50  * the data entry block. Gets updated to the start of the next row
51  * or the end of the data block, if the data block is the last one on
52  * that row.
53  *
54  * \param[in] column_modes The modes of each column.
55  *
56  * \param[in] has_target Whether the row starts with the target value
57  * or not.
58  *
59  * \param[in] column_index_offset When filling in the ml_data_entry
60  * structures, this gives the starting index that the column_index
61  * value is based off of.
62  */
63 template <typename EntryContainer>
64 GL_HOT_INLINE_FLATTEN
65 static inline void copy_raw_into_ml_data_entry_row(
66  EntryContainer& row,
67  const row_metadata& rm,
68  entry_value_iterator row_block_ptr,
69  const std::shared_ptr<ml_data_side_features>& side_features) {
70 
71  typedef typename EntryContainer::value_type Entry;
72 
73  DASSERT_TRUE(row.empty());
74 
75  size_t write_index = 0;
76 
77  read_ml_data_row(
78  rm,
79 
80  row_block_ptr,
81 
82  /** The function to write out the data to x.
83  */
84  [&](ml_column_mode mode, size_t column_index,
85  size_t feature_index, double value,
86  size_t index_size, size_t index_offset) GL_GCC_ONLY(GL_HOT_INLINE_FLATTEN) {
87 
88  size_t global_index = (LIKELY(feature_index < index_size)
89  ? index_offset + feature_index
90  : size_t(-1));
91 
92  Entry e;
93  e = ml_data_full_entry{column_index, feature_index, global_index, value};
94  _add_element(row, write_index, e);
95  ++write_index;
96  },
97 
98  // Nothing that we need to do at the end of each column.
99  [&](ml_column_mode, size_t, size_t) {},
100 
101  side_features);
102 }

104 // The main function that implements all of the above filling
105 // techniques.
106 template <typename SparesOrDenseEigenVector>
107 GL_HOT_INLINE_FLATTEN
108 inline void copy_raw_into_eigen_array(
109  SparesOrDenseEigenVector& x,
110  const row_metadata& rm,
111  const entry_value_iterator& row_block_ptr,
112  const std::shared_ptr<ml_data_side_features>& side_features,
113  bool use_reference) {
114 
115  size_t offset = 0;
116 
117  read_ml_data_row(
118  /** The row metadata. **/
119  rm,
120 
121  /** The pointer to the current location.
122  */
123  row_block_ptr,
124 
125  /** The function to write out the data to x.
126  */
127  [&](ml_column_mode mode, size_t column_index,
128  size_t feature_index, double value,
129  size_t index_size, size_t index_offset) {
130 
131  if(UNLIKELY(feature_index >= index_size))
132  return;
133 
134  size_t idx = offset + feature_index;
135 
136  // Decrement the category if it isn't the reference category.
137  if(use_reference && mode_is_categorical(mode)) {
138  if (feature_index != 0) {
139  idx -= 1;
140  } else {
141  return;
142  }
143  }
144 
145  DASSERT_GE(idx, 0);
146  DASSERT_LT(idx, size_t(x.size()));
147  x.coeffRef(idx) = value;
148 
149  },
150 
151  /** The function to advance the offset, called after each column
152  * is finished.
153  */
154  [&](ml_column_mode mode, size_t column_index,
155  size_t index_size) GL_GCC_ONLY(GL_HOT_INLINE_FLATTEN) {
156  offset += (index_size
157  - ((use_reference && mode_is_categorical(mode)) ? 1 : 0));
158  },
159 
160  /** The pointer to the side features. Possibly null.
161  */
162  side_features);
163 }
